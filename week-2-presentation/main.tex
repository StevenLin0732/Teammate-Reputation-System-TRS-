\begin{frame}{Intro: Backend}
	\begin{itemize}
		\item \textbf{Framework:} Flask (Python) with server-side rendered pages (Jinja2 templates)
		\item \textbf{Data layer:} Flask-SQLAlchemy (SQLAlchemy ORM)
		\item \textbf{Database:} SQLite (local file DB, e.g., \texttt{teamrank.db})
		\item \textbf{Models / tables:} User, Lobby, Team, TeamMember, Submission, Rating
		\item \textbf{How data access works:} Python objects \textrightarrow{} ORM queries \textrightarrow{} generated SQL
		\item \textbf{Notes:} simple schema evolution for SQLite (no full migration tool yet)
	\end{itemize}
